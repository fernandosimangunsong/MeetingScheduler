 public void listUser() throws IOException{
        long id;
        String name = "";
	String contact = "";		
        
        ObjectMapper mapper = new ObjectMapper();
        JsonNode root = mapper.readTree(new File("coba.json"));
        // Get Contact
			JsonNode contactNode = root.path("contact");
			

			for (JsonNode node : contactNode) {
				String type = node.path("type").asText();
				String ref = node.path("ref").asText();
				System.out.println("type : " + type);
				System.out.println("ref : " + ref);
			} 
                
        JsonNode users = mapper.readTree(new File("users1.json"));
          
			JsonNode user = users.path("users");
                        
	 for (JsonNode node : user) {
				String email = node.path("email").asText();
				String nama = node.path("name").asText();
				System.out.println("email : " + email);
				System.out.println("nama : " + nama);
	} 


 public List bacaFileUser() throws IOException{
        ObjectMapper mapper = new ObjectMapper();
        List datauser = new ArrayList<>();
        datauser = mapper.readValue(new File("user.json"),List.class);
        return datauser;
        
    }
    
    
    
    public void simpan(List datauser) throws IOException  {
	ObjectMapper mapper = new ObjectMapper();
        mapper.writerWithDefaultPrettyPrinter().writeValue(new File("user.json"),datauser);
    }
